char * strInsert(char *strDest, int n, const char *strIn)
/* ?strIn???strDest[n]?? */
{
	int l = strlen(strIn);
	int i;
	for (i=strlen(strDest); i>n; i--)
		strDest[i+l] = strDest[i];
	for (i=0; i<l; i++)
		strDest[n+1+i] = strIn[i];
	return strDest;
}

{
	char str[15], substr[4];
